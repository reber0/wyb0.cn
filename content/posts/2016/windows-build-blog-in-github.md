### 0x03 配置config.toml
```
# 将reber全部替换为你github的账户名，另外你也可以进行一些基本的设置
baseurl = "http://reber.github.io"
languageCode = "en-us"
title = "reber's Blog"
theme = "beautifulhugo"
pygmentsUseClasses = false
#disqusShortname = "XXX"
#googleAnalytics = "XXX"

[Params]
  logo = "img/avatar-icon.png"
  subtitle = "This is my personal blog"

[Author]
  name = "Some Person"
  email = "123123123@163.com"
  github = "reber"

[[menu.main]]
    name = "Blog"
    url = "/"
    weight = 1

[[menu.main]]
    name = "About"
    url = "/page/about/"
    weight = 2
```
